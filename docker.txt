open command prompt 
docker --version          #to check docker
docker run hello-world    #to pull helloworld command first
docker images             #it will show all the images present in a docker
docker pull ubuntu        #pull ubuntu image from docker hub
docker run -it -d ubuntu  #run the image and create a container of the image
docker ps -a              #this will show all the containers
docker exec -it <docker_id> bash #this will be used to accessing and running container
#enter echo hello in that particular container 
# enter exit to come out of the container
docker stop <docker_id>   #to stop the running container
docker commit <docker_id> <dockerhub username>/ubuntu # this will create new image in ubuntu docker hub container
 docker push <username>/ubuntu   #push image to docker hub
  
 docker rm <containr_id>  #to delete the container
 docker rmi <image id>    #to delete image
